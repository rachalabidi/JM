243 public void paint(Graphics g) {
244
245 int cell = 0;
246 int uncover = 0;
247
248
249 for (int i = 0; i < rows; i++) {
250 for (int j = 0; j < cols; j++) {
251
252 cell = field[(i * cols) + j];
253
254 if (inGame && cell == MINE_CELL)
255 inGame = false;
256
257 if (!inGame) {
258 if (cell == COVERED_MINE_CELL) {
259 cell = DRAW_MINE;
260 } else if (cell == MARKED_MINE_CELL) {
261 cell = DRAW_MARK;
262 } else if (cell > COVERED_MINE_CELL) {
263 cell = DRAW_WRONG_MARK;
264 } else if (cell > MINE_CELL) {
265 cell = DRAW_COVER;
266 }
267
268
269 } else {
270 if (cell > COVERED_MINE_CELL)
271 cell = DRAW_MARK;
272 else if (cell > MINE_CELL) {
273 cell = DRAW_COVER;
274 uncover++;
275 }
276 }
277
278 g.drawImage(img[cell], (j * CELL_SIZE),
279 (i * CELL_SIZE), this);
280 }
281 }
282
283
284 if (uncover == 0 && inGame) {
285 inGame = false;
286 statusbar.setText("Game won");
287 } else if (!inGame)
288 statusbar.setText("Game lost");
289 }

322 mines_left--;
